function doGet(e) {
  try {
    const action = e.parameter.action;

    if (action === "getOnlineUsers") {
      return jsonResponse({
        success: true,
        users: getOnlineUsers()
      });
    }

    if (action === "getVisitLogs") {
      return jsonResponse({
        success: true,
        logs: getVisitLogs(Number(e.parameter.limit || 50))
      });
    }

    if (action === "getOrdersByUid") {
      return jsonResponse({
        success: true,
        orders: getOrdersByUid(e.parameter.uid)
      });
    }

    if (action === "getOrderMessages") {
      return jsonResponse({
        success: true,
        messages: getOrderMessages(e.parameter.orderId)
      });
    }

    if (action === "getPublicChatThreads") {
      return jsonResponse({
        success: true,
        threads: getPublicChatThreads()
      });
    }

    if (action === "getPublicChatMessages") {
      return jsonResponse({
        success: true,
        messages: getPublicChatMessages(e.parameter.threadId, e.parameter.email)
      });
    }

    if (action === "getAllOrders") {
      return jsonResponse({
        success: true,
        orders: getAllOrders()
      });
    }

    return jsonResponse({
      success: false,
      message: "Unknown action"
    });
  } catch (err) {
    return jsonResponse({
      success: false,
      message: err.message
    });
  }
}

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents || "{}");

    if (data.action === "trackPresence") {
      return jsonResponse(trackPresence(data));
    }

    if (data.action === "trackVisit") {
      return jsonResponse(trackVisit(data));
    }

    if (data.action === "createOrder") {
      return jsonResponse(createOrder(data));
    }

    if (data.action === "sendOrderMessage") {
      return jsonResponse(sendOrderMessage(data));
    }

    if (data.action === "sendPublicChatMessage") {
      return jsonResponse(sendPublicChatMessage(data));
    }

    if (data.action === "updateOrderByAdmin") {
      return jsonResponse(updateOrderByAdmin(data));
    }

    return jsonResponse({
      success: false,
      message: "Unknown action"
    });
  } catch (err) {
    return jsonResponse({
      success: false,
      message: err.message
    });
  }
}

function setupOrdersSheet() {
  const sheet = getSheet();
  const firstRow = sheet.getRange(1, 1, 1, HEADERS.length).getValues()[0];
  const hasHeaders = firstRow.some(value => value);

  if (!hasHeaders) {
    sheet.getRange(1, 1, 1, HEADERS.length).setValues([HEADERS]);
    sheet.setFrozenRows(1);
  }
}

function setupSheetHeaders(sheet, headers) {
  const firstRow = sheet.getRange(1, 1, 1, headers.length).getValues()[0];
  const hasHeaders = firstRow.some(value => value);

  if (!hasHeaders) {
    sheet.getRange(1, 1, 1, headers.length).setValues([headers]);
    sheet.setFrozenRows(1);
  }
}

function createOrder(data) {
  const sheet = getSheet();
  setupOrdersSheet();

  const now = new Date();
  const orderId = data.orderId || createOrderId();
  const slipImage = data.image ? uploadImage(data.image, orderId) : "";

  sheet.appendRow([
    now,
    orderId,
    data.uid || "",
    data.email || "",
    data.customerName || data.name || "",
    data.item || "",
    data.setNumber || "",
    data.amount || "",
    data.note || "",
    slipImage,
    data.status || "รอตรวจสอบ",
    data.adminReply || "",
    data.code || "",
    now
  ]);

  return {
    success: true,
    orderId: orderId
  };
}

function getOrdersByUid(uid) {
  if (!uid) return [];

  return getAllOrders()
    .filter(order => order.uid === uid)
    .sort((a, b) => new Date(b.createdAt) - new Date(a.createdAt));
}

function getAllOrders() {
  const sheet = getSheet();
  setupOrdersSheet();

  const values = sheet.getDataRange().getValues();
  const rows = values.slice(1);

  return rows
    .filter(row => row[1])
    .map(rowToOrder)
    .sort((a, b) => new Date(b.createdAt) - new Date(a.createdAt));
}

function sendOrderMessage(data) {
  const sheet = getChatSheet();
  setupSheetHeaders(sheet, CHAT_HEADERS);

  if (!data.orderId) {
    return {
      success: false,
      message: "Missing orderId"
    };
  }

  if (!data.message) {
    return {
      success: false,
      message: "Missing message"
    };
  }

  sheet.appendRow([
    new Date(),
    data.orderId || "",
    data.senderType || "customer",
    data.uid || "",
    data.email || "",
    data.displayName || "",
    data.message || ""
  ]);

  return {
    success: true
  };
}

function getOrderMessages(orderId) {
  if (!orderId) return [];

  const sheet = getChatSheet();
  setupSheetHeaders(sheet, CHAT_HEADERS);

  const values = sheet.getDataRange().getValues();
  const rows = values.slice(1);

  return rows
    .filter(row => row[1] === orderId)
    .map(row => ({
      createdAt: row[0],
      orderId: row[1],
      senderType: row[2],
      uid: row[3],
      email: row[4],
      displayName: row[5],
      message: row[6]
    }))
    .sort((a, b) => new Date(a.createdAt) - new Date(b.createdAt));
}

function sendPublicChatMessage(data) {
  const sheet = getPublicChatSheet();
  setupSheetHeaders(sheet, PUBLIC_CHAT_HEADERS);

  const normalizedEmail = normalizeEmail(data.email);
  const threadId = data.threadId || createPublicThreadId(normalizedEmail);

  if (!normalizedEmail) {
    return {
      success: false,
      message: "Missing email"
    };
  }

  if (!data.name) {
    return {
      success: false,
      message: "Missing name"
    };
  }

  if (!data.message) {
    return {
      success: false,
      message: "Missing message"
    };
  }

  sheet.appendRow([
    new Date(),
    threadId,
    data.senderType || "customer",
    data.name || "",
    normalizedEmail,
    data.message || ""
  ]);

  return {
    success: true,
    threadId: threadId
  };
}

function getPublicChatMessages(threadId, email) {
  const sheet = getPublicChatSheet();
  setupSheetHeaders(sheet, PUBLIC_CHAT_HEADERS);

  const resolvedThreadId = threadId || createPublicThreadId(normalizeEmail(email));
  if (!resolvedThreadId) return [];

  const values = sheet.getDataRange().getValues();
  const rows = values.slice(1);

  return rows
    .filter(row => row[1] === resolvedThreadId)
    .map(row => ({
      createdAt: row[0],
      threadId: row[1],
      senderType: row[2],
      name: row[3],
      email: row[4],
      message: row[5]
    }))
    .sort((a, b) => new Date(a.createdAt) - new Date(b.createdAt));
}

function getPublicChatThreads() {
  const sheet = getPublicChatSheet();
  setupSheetHeaders(sheet, PUBLIC_CHAT_HEADERS);

  const values = sheet.getDataRange().getValues();
  const rows = values.slice(1).filter(row => row[1]);
  const threadMap = {};

  rows.forEach(row => {
    const threadId = row[1];
    const createdAt = row[0];

    if (!threadMap[threadId] || new Date(createdAt) > new Date(threadMap[threadId].createdAt)) {
      threadMap[threadId] = {
        createdAt: createdAt,
        threadId: threadId,
        senderType: row[2],
        name: row[3],
        email: row[4],
        message: row[5]
      };
    }
  });

  return Object.keys(threadMap)
    .map(key => threadMap[key])
    .sort((a, b) => new Date(b.createdAt) - new Date(a.createdAt));
}

function trackPresence(data) {
  const sheet = getPresenceSheet();
  setupSheetHeaders(sheet, PRESENCE_HEADERS);

  if (!data.uid) {
    return {
      success: false,
      message: "Missing uid"
    };
  }

  const values = sheet.getDataRange().getValues();
  const now = new Date();
  const lastSeen = now;

  for (let i = 1; i < values.length; i++) {
    if (values[i][0] === data.uid) {
      const rowIndex = i + 1;
      sheet.getRange(rowIndex, 2).setValue(data.email || "");
      sheet.getRange(rowIndex, 3).setValue(data.displayName || "");
      sheet.getRange(rowIndex, 4).setValue(data.page || "");
      sheet.getRange(rowIndex, 5).setValue(data.pageLabel || "");
      sheet.getRange(rowIndex, 6).setValue(lastSeen);
      sheet.getRange(rowIndex, 7).setValue(data.lastAction || "active");
      sheet.getRange(rowIndex, 8).setValue(now);

      return { success: true };
    }
  }

  sheet.appendRow([
    data.uid,
    data.email || "",
    data.displayName || "",
    data.page || "",
    data.pageLabel || "",
    lastSeen,
    data.lastAction || "active",
    now
  ]);

  return { success: true };
}

function getOnlineUsers() {
  const sheet = getPresenceSheet();
  setupSheetHeaders(sheet, PRESENCE_HEADERS);

  const values = sheet.getDataRange().getValues();
  const rows = values.slice(1);
  const cutoff = Date.now() - ONLINE_ACTIVE_MINUTES * 60 * 1000;

  return rows
    .filter(row => row[0] && row[5] && new Date(row[5]).getTime() >= cutoff)
    .map(row => ({
      uid: row[0],
      email: row[1],
      displayName: row[2],
      page: row[3],
      pageLabel: row[4],
      lastSeen: row[5],
      lastAction: row[6],
      updatedAt: row[7]
    }))
    .sort((a, b) => new Date(b.lastSeen) - new Date(a.lastSeen));
}

function getVisitLogs(limit) {
  const sheet = getVisitLogSheet();
  setupSheetHeaders(sheet, VISIT_LOG_HEADERS);

  const safeLimit = Math.max(1, Math.min(Number(limit) || 50, 200));
  const values = sheet.getDataRange().getValues();
  const rows = values.slice(1).filter(row => row[0]);

  return rows
    .map(row => ({
      createdAt: row[0],
      uid: row[1],
      email: row[2],
      displayName: row[3],
      eventType: row[4],
      page: row[5],
      pageLabel: row[6],
      note: row[7]
    }))
    .sort((a, b) => new Date(b.createdAt) - new Date(a.createdAt))
    .slice(0, safeLimit);
}

function updateOrderByAdmin(data) {
  const sheet = getSheet();
  const values = sheet.getDataRange().getValues();
  const orderId = data.orderId;

  if (!orderId) {
    return {
      success: false,
      message: "Missing orderId"
    };
  }

  for (let i = 1; i < values.length; i++) {
    if (values[i][1] === orderId) {
      const rowIndex = i + 1;
      sheet.getRange(rowIndex, 11).setValue(data.status || values[i][10]);
      sheet.getRange(rowIndex, 12).setValue(data.adminReply || "");
      sheet.getRange(rowIndex, 14).setValue(new Date());

      return {
        success: true,
        orderId: orderId
      };
    }
  }

  return {
    success: false,
    message: "Order not found"
  };
}

function uploadImage(base64Data, orderId) {
  try {
    const base64 = base64Data.split(",")[1];
    const blob = Utilities.newBlob(
      Utilities.base64Decode(base64),
      "image/jpeg",
      "slip_" + orderId + "_" + Date.now() + ".jpg"
    );

    const folder = getSlipFolder();
    const file = folder.createFile(blob);
    return file.getUrl();
  } catch (err) {
    Logger.log("UPLOAD ERROR: " + err.message);
    return "";
  }
}

function getSlipFolder() {
  try {
    return DriveApp.getFolderById(SLIP_FOLDER_ID);
  } catch (err) {
    Logger.log("GET FOLDER BY ID ERROR: " + err.message);
  }

  const folders = DriveApp.getFoldersByName(SLIP_FOLDER_NAME);

  if (folders.hasNext()) {
    return folders.next();
  }

  return DriveApp.createFolder(SLIP_FOLDER_NAME);
}

function rowToOrder(row) {
  return {
    createdAt: row[0],
    orderId: row[1],
    uid: row[2],
    email: row[3],
    customerName: row[4],
    item: row[5],
    setNumber: row[6],
    amount: row[7],
    note: row[8],
    slipImage: row[9],
    status: row[10],
    adminReply: row[11],
    code: row[12],
    updatedAt: row[13]
  };
}

function createOrderId() {
  return "ORDER-" + Utilities.formatDate(new Date(), "Asia/Bangkok", "yyyyMMdd-HHmmss") + "-" + Math.floor(Math.random() * 1000);
}

function getSheet() {
  return SpreadsheetApp
    .openById(SPREADSHEET_ID)
    .getSheetByName(SHEET_NAME);
}

function getPresenceSheet() {
  return getOrCreateSheet(PRESENCE_SHEET_NAME);
}

function getVisitLogSheet() {
  return getOrCreateSheet(VISIT_LOG_SHEET_NAME);
}

function getChatSheet() {
  return getOrCreateSheet(CHAT_SHEET_NAME);
}

function getPublicChatSheet() {
  return getOrCreateSheet(PUBLIC_CHAT_SHEET_NAME);
}

function getOrCreateSheet(name) {
  const spreadsheet = SpreadsheetApp.openById(SPREADSHEET_ID);
  let sheet = spreadsheet.getSheetByName(name);

  if (!sheet) {
    sheet = spreadsheet.insertSheet(name);
  }

  return sheet;
}

function normalizeEmail(email) {
  return String(email || "").trim().toLowerCase();
}

function createPublicThreadId(email) {
  if (!email) return "";
  return "public-" + email.replace(/[^a-z0-9@._-]/g, "_");
}

function jsonResponse(payload) {
  return ContentService
    .createTextOutput(JSON.stringify(payload))
    .setMimeType(ContentService.MimeType.JSON);
}
